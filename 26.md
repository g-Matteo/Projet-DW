* Vue matérialisée 1 : MV_Performance_Commerciale
```sql
CREATE MATERIALIZED VIEW MV_Performance_Commerciale
BUILD IMMEDIATE
REFRESH COMPLETE ON DEMAND
ENABLE QUERY REWRITE
AS
SELECT 
    D.Mois_Annee,
    D.Trimestre,
    D.Annee,
    G.Pays,
    A.Nom_Plan,
    E.Categorie_Evenement,
    SUM(F.Montant_Facture) AS Total_Facture,
    SUM(F.Delta_MRR) AS Total_Delta_MRR,
    COUNT(*) AS Nb_Evenements
FROM Fait_Evenement_Abonnement F
JOIN Dim_Date D ON F.Date_Key = D.Date_Key
JOIN Dim_Geographie G ON F.Geo_Key = G.Geo_Key
JOIN Dim_Abonnement A ON F.Abonnement_Key = A.Abonnement_Key
JOIN Dim_Type_Evenement E ON F.Evenement_Key = E.Evenement_Key
GROUP BY D.Mois_Annee, D.Trimestre, D.Annee, G.Pays, A.Nom_Plan, E.Categorie_Evenement;
```

* Vue matérialisée 2 : MV_Analyse_Profil
```sql
CREATE MATERIALIZED VIEW MV_Analyse_Profil
BUILD IMMEDIATE
REFRESH COMPLETE ON DEMAND
ENABLE QUERY REWRITE
AS
SELECT 
    D.Trimestre,
    D.Annee,
    P.Tranche_Age,
    P.Categorie_socio_pro,
    E.Categorie_Evenement,
    SUM(F.Montant_Facture) AS Total_Facture,
    COUNT(*) AS Nb_Evenements
FROM Fait_Evenement_Abonnement F
JOIN Dim_Date D ON F.Date_Key = D.Date_Key
JOIN Dim_Profil P ON F.Profil_Key = P.Profil_Key
JOIN Dim_Type_Evenement E ON F.Evenement_Key = E.Evenement_Key
GROUP BY D.Trimestre, D.Annee, P.Tranche_Age, P.Categorie_socio_pro, E.Categorie_Evenement;
```
